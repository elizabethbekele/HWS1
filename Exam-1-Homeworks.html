  1. Make a function that does 9, 10, and 11 together for you, and returns the number of doubles. (Note: functions dont have to have input arguments!)
set.seed(102389)

combo <- function(){
  fifty_rolls_1 <- sample(1:6, 100, replace = T)
  fifty_rolls_2 <- sample(1:6, 100, replace = T)
  return (length(fifty_rolls_1[fifty_rolls_1 == fifty_rolls_2]))
}

combo()
